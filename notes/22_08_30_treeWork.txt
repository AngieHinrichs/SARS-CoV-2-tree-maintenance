# (past) Time to mask 28877 and 28878 as discussed in
# https://github.com/cov-lineages/pango-designation/issues/988
# I've also noticed some UTR5 bases with a lot of noise causing some mini-BAs, sometimes
# attracting a lot of sequences from countries with not too many sequences overall,
# so update maskDelta.sh to mask more of those too.

today=2022-08-29
cd $ottoDir/$today

time $scriptDir/maskDelta.sh merged.pb merged.deltaMasked.pb
#real    11m40.582s

# Prune samples with too many private mutations and internal branches that are too long.
$matUtils extract -i merged.deltaMasked.pb \
    --max-parsimony 20 \
    --max-branch-length 60 \
    --max-path-length 175 \
    -O -o merged.deltaMasked.filtered.pb

# Two rounds of matOptimize since that's necessary after introducing new masking.
time $matOptimize \
    -T 60 -r 8 -M 2 -S move_log.opt1 \
    -i merged.deltaMasked.filtered.pb \
    -o merged.deltaMasked.filtered.opt1.pb \
    >& matOptimize.opt1.log

time $matOptimize \
    -T 60 -r 8 -M 2 -S move_log.opt2 \
    -i merged.deltaMasked.filtered.opt1.pb \
    -o gisaidAndPublic.$today.masked.preTrim.pb \
    >& matOptimize.opt2.log

| g -Fwf - idToName \

wc -l shortNameGbDupsToRemove

# Now update names.  Use the latest latest metadata (not just old $today).
g ^EPI_ISL idToName > epiToOldName
cut -f 1 epiToOldName \
| grep -Fwf - <(zcat $gisaidDir/metadata_batch_$(date +%F).tsv.gz) \
| tawk '{print $3, $1 "|" $3 "|"$5;}' \
| sort > epiToNewName
join -t$'\t' epiToOldName epiToNewName \
| tawk '$2 != $3 {print $2, $3;}' \
    > epiOldToNewName
wc -l epiOldToNewName
#6666 epiOldToNewName

egrep '^[A-Z]{2}[0-9]{6}\.[0-9]+' idToName > gbAccToOldName
# Wow, we're really getting GB names from fasta (with metadata stuff plopped in) not metadata...
# (adapted from makeNewMaskedVcf.sh):
time fastaNames $ncbiDir/genbank.fa.xz > gbFastaNames
grep COG-UK/ gbFastaNames \
| sed -re 's/[ |].*//' \
| grep -Fwf - $ncbiDir/ncbi_dataset.plusBioSample.tsv \
| tawk '{print $1, $4 "/" $6 "/" $3 "|" $1 "|" $3;}' \
| sed -re 's@COG-UK/@@g; s/United Kingdom://; s/(\/[0-9]{4})(-[0-9]+)*/\1/; s/ //g;' \
    > gbAccToNewName
grep -v COG-UK/ gbFastaNames \
| sed -re 's/[ |].*//' \
| grep -Fwf - $ncbiDir/ncbi_dataset.plusBioSample.tsv \
| tawk '{ if ($3 == "") { $3 = "?"; }
          if ($6 != "") { print $1 "\t" $6 "|" $1 "|" $3; }
          else { print $1 "\t" $1 "|" $3; } }' \
| cleanGenbank \
| sed -re 's/ /_/g' \
    >> gbAccToNewName
join -t$'\t' gbAccToOldName <(sort gbAccToNewName) \
| tawk '$2 != $3 {print $2, $3;}' \
    > gbOldToNewName
wc -l gbOldToNewName
#74863 gbOldToNewName

cat epiOldToNewName gbOldToNewName > oldToNewName

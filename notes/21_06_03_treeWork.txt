# We have a bunch of outdated names in our tree, some with the wrong year.
# Update those names and remove withdrawn sequences while we're at it.

buildDate=2021-06-02
# Repeated 2021-06-07 with buildDate=2021-06-04 after first build success in a few days (segv fixed)
cd /hive/data/outside/otto/sarscov2phylo/$buildDate
grep EPI_ISL samples.$buildDate \
| awk -F\| '{print $2 "\t" $0;}' \
| sort > epiToCurrentTreeName
zcat $gisaidDir/metadata_batch_$buildDate.tsv.gz \
| grep -Fwf <(cut -f 1 epiToCurrentTreeName) \
| tawk '{print $3, $1 "|" $3 "|" $5;}' \
| sort > epiToLatestName
wc -l epiToCurrentTreeName epiToLatestName
#  1070576 epiToCurrentTreeName
#  1060018 epiToLatestName

# Prune withdrawn sequences from tree
comm -23 <(cut -f 1 epiToCurrentTreeName ) <(cut -f 1 epiToLatestName ) > epis.withdrawn
wc -l epis.withdrawn
#10558 epis.withdrawn
grep -Fwf epis.withdrawn epiToCurrentTreeName \
| cut -f 2 > epis.withdrawn.treeNames
$matUtils extract \
    -p \
    -i gisaidAndPublic.$buildDate.masked.pb \
    -s epis.withdrawn.treeNames \
    -O -o gisaidAndPublic.$buildDate.masked.withdrawnPruned.pb
$matUtils summary -i gisaidAndPublic.$buildDate.masked.pb
#Total Nodes in Tree: 2328173
#Total Samples in Tree: 1903213
#Total Condensed Nodes in Tree: 234449
#Total Samples in Condensed Nodes: 1224430
#Total Tree Parsimony: 1832406
$matUtils summary -i gisaidAndPublic.$buildDate.masked.withdrawnPruned.pb
#Total Nodes in Tree: 2314269
#Total Samples in Tree: 1892655
#Total Condensed Nodes in Tree: 230717
#Total Samples in Condensed Nodes: 1210542
#Total Tree Parsimony: 1831830
expr 1903213 - 1892655
#10558

# Update names that have been changed in GISAID
join -t$'\t' epiToCurrentTreeName epiToLatestName \
| tawk '$2 != $3 {print $2, $3;}' \
    > gisaidNameChanges
wc -l gisaidNameChanges
#24383 gisaidNameChanges
$matUtils mask -T 50 \
    -i gisaidAndPublic.$buildDate.masked.withdrawnPruned.pb \
    -r gisaidNameChanges \
    -o gisaidAndPublic.$buildDate.masked.namesUpdated.pb >& rename.out
$matUtils summary -i gisaidAndPublic.$buildDate.masked.namesUpdated.pb
#Total Nodes in Tree: 2314269
#Total Samples in Tree: 1892655
#Total Condensed Nodes in Tree: 0
#Total Samples in Condensed Nodes: 0
#Total Tree Parsimony: 1831830
# Wow, no condensing. Well... I guess matUtils extract then.
$matUtils extract \
    -i gisaidAndPublic.$buildDate.masked.namesUpdated.pb \
    -O -o gisaidAndPublic.$buildDate.masked.namesUpdated.condensed.pb
$matUtils summary -i gisaidAndPublic.$buildDate.masked.namesUpdated.condensed.pb
#Total Nodes in Tree: 2314269
#Total Samples in Tree: 1892655
#Total Condensed Nodes in Tree: 230717
#Total Samples in Condensed Nodes: 1210542
#Total Tree Parsimony: 1831830
# OK, now we're square.  It is too late for this to be used for the next build, but
# I know what to do next time a build finishes at a reasonable hour.

# 2021-06-07: replace main pb so it's used as base for today's build.
rm gisaidAndPublic.$buildDate.masked.pb
ln gisaidAndPublic.$buildDate.masked.namesUpdated.condensed.pb gisaidAndPublic.$buildDate.masked.pb
# 2021-06-07 done at this point.

# Now see how well we match up with names used in pango-designation/lineages.csv,
# since lineages.metadata.csv is out of date & missing a handful of lineages.
$matUtils extract -i gisaidAndPublic.$buildDate.masked.namesUpdated.condensed.pb \
    -u samples.$buildDate.postUpdate

tail -n+2 ~/github/pango-designation/lineages.csv \
| tr ',' $'\t' \
| sed -re 's/^([A-Za-z]+)_/\1/; s/^([A-Za-z]+)_/\1/; s/^([A-Za-z]+)_/\1/;' \
| sort > pangoNameNo_ToLineage
awk -F\| '{print $1 "\t" $0;}' samples.$buildDate.postUpdate \
| sed -re 's/^([A-Za-z]+)_/\1/;' \
| sort > nameNo_ToTreeName
comm -23 <(cut -f1 pangoNameNo_ToLineage) <(cut -f1 nameNo_ToTreeName) | wc -l
#12719
join -t$'\t' pangoNameNo_ToLineage nameNo_ToTreeName \
| cut -f 2,3 \
| sort > lineageToName.postUpdate
wc -l pangoNameNo_ToLineage lineageToName.postUpdate
#  457148 pangoNameNo_ToLineage
#  456987 lineageToName.postUpdate
comm -23 <(cut -f1 pangoNameNo_ToLineage) <(cut -f1 nameNo_ToTreeName) > pangoNoTree
# Never mind the ones that don't appear in lineages.metadata.csv because they're outdated:
comm -23 \

# Make a pangolin-only annotated protobuf (pangolin v3 --usher does not handle protobufs with multiple annotations)

    -l \
    -i gisaidAndPublic.$buildDate.masked.namesUpdated.condensed.pb \
    -c lineageToName.updated \
    -u mutations.pangolin \
    -D details.pangolin \
    -o gisaidAndPublic.$buildDate.masked.namesUpdated.pangolin.pb \
    >& annotate.pangolin.namesUpdated.out

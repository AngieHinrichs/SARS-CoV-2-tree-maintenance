# As requested by Chris in https://github.com/cov-lineages/pango-designation/issues/435 , mask
# 28254 in Delta, optimize and see if AY.96 is absorbed into AY.46.

today=2022-02-15
cd $ottoDir/$today
time $matUtils extract -i gisaidAndPublic.$today.masked.pb -C clade-paths -S sample-paths
deltaNode=$(grep IND/GBRC714b/2021 sample-paths | awk '{print $NF;}' | sed -re 's/:.*//;')
echo -e "N28254N\t$deltaNode" > mask28254.tsv

time $matUtils mask -i gisaidAndPublic.$today.masked.pb \
    -m mask28254.tsv \
    -o gisaidAndPublic.$today.masked.delta28254.pb
#real    2m0.970s

$matOptimize -T 50 -r 8 -M 1 \
    -S move_log.delta28254 -i gisaidAndPublic.$today.masked.delta28254.pb \
    -o gisaidAndPublic.$today.masked.delta28254.opt.pb \
    >& matOptimize.delta28254.log
#original parsimony score:6685534
#parsimony score after optimizing: 6685287,with radius 8, second from start 1164 
#parsimony score after optimizing: 6685286,with radius 8, second from start 1185 
#parsimony score after optimizing: 6685286,with radius 8, second from start 1204 
#parsimony score after optimizing: 6685286,with radius 8, second from start 1222 
#...
#parsimony score after optimizing: 6685286,with radius 8, second from start 1803 
#Final Parsimony score 6685286

# Make a taxonium protobuf:
zcat /hive/data/genomes/wuhCor1/goldenPath/bigZips/genes/ncbiGenes.gtf.gz \
| grep -v '"ORF1a"' > ncbiGenes.gtf
zcat /hive/data/genomes/wuhCor1/wuhCor1.fa.gz > wuhCor1.fa
zcat gisaidAndPublic.$today.metadata.tsv.gz > metadata.tmp.tsv
time $matUtils extract -i gisaidAndPublic.$today.masked.delta28254.opt.pb \
    -f wuhCor1.fa \
    -g ncbiGenes.gtf \
    -M metadata.tmp.tsv \
    --extra-fields pango_lineage_usher \
    --write-taxodium gisaidAndPublic.$today.masked.delta28254.opt.taxodium.pb
rm metadata.tmp.tsv wuhCor1.fa
gzip -f gisaidAndPublic.$today.masked.delta28254.opt.taxodium.pb

# That wasn't enough -- half of the former AY.96 is held back by a back-mutation at 28461 (N:63).
# And coloring by N:63, there seems to be a lot of counterproductive flip-flopping in Delta.
# So, I guess we better mask that too.
echo -e "N28254N\t$deltaNode" > maskTwo.tsv
echo -e "N28461N\t$deltaNode" >> maskTwo.tsv

time $matUtils mask -i gisaidAndPublic.$today.masked.pb \
    -m maskTwo.tsv \
    -o gisaidAndPublic.$today.masked.deltaTwo.pb

    -o gisaidAndPublic.$today.masked.deltaTwo.opt.pb \
    >& matOptimize.deltaTwo.log

mv move_log.deltaTwo{,.wedged}

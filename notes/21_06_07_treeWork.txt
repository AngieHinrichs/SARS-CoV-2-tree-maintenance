# The daily build process was disrupted by a segv in usher when writing clades.txt
# (after ~18 hours of building up the big tree) that took a few days to debug and
# confirm the fix, so this is a good point to clean up withdrawn sequences, outdated
# names, and duplicates before restarting the build.

# First I went through the steps in 2021_06_03_treeWork.txt to remove withdrawn GISAID
# seqs and update GISAID names.  Should do that for GenBank as well; also remove items
# that are duplicated because of some shortened dates (ugh, what a dumb idea).

buildDate=2021-06-07

# Remove date-dups.
$matUtils extract -i gisaidAndPublic.$buildDate.masked.pb -u samples.$buildDate
awk -F\| '{ if ($3 == "") { print $1; } else { print $1 "|" $2; } }' samples.$buildDate \
| sort > noDateNames
uniq -c noDateNames | awk '$1 > 1 {print $2;}' \
| grep -Fwf - samples.$buildDate \
| grep -E '\|[0-9][0-9](-[0-9]+)*$' \
    > dateDupsToRemove
wc -l dateDupsToRemove 
#5418 dateDupsToRemove
$matUtils extract -i gisaidAndPublic.$buildDate.masked.pb \
    -p -s dateDupsToRemove \
    -o gisaidAndPublic.$buildDate.masked.noDateDups.pb
$matUtils extract -i gisaidAndPublic.$buildDate.masked.noDateDups.pb -u samples.$buildDate

# NOTE FOR NExT TIME: this left out nameless items like LC528232.2|20-02-10
awk -F\| '$3 != "" && $2 ~ /^[A-Z]{2}[0-9{6}\.[0-9]+$/ {print $2 "\t" $0;}' samples.$buildDate \
| sort > gbToOldName
wc -l gbToOldName
#523730 gbToOldName

# Remove withdrawn GenBank seqs and update names.
# Special naming for COG-UK sequences: strip COG-UK/, add back country and year
grep COG-UK/ $ncbiDir/ncbi_dataset.plusBioSample.tsv \
| tawk '{print $1, $4 "/" $6 "/" $3 "|" $1 "|" $3;}' \
| sed -re 's@COG-UK/@@g; s/United Kingdom://; s/(\/[0-9]{4})(-[0-9]+)*/\1/; s/ //g;' \
    > tmp
grep -v COG-UK/ $ncbiDir/ncbi_dataset.plusBioSample.tsv \
| grep -E '^[A-Z]{2}[0-9{6}\.[0-9]+' \
| tawk '{ if ($3 == "") { $3 = "?"; }
          if ($6 != "") { print $1 "\t" $6 "|" $1 "|" $3; }
          else { print $1 "\t" $1 "|" $3; } }' \
| cleanGenbank \
| sed -re 's/ /_/g' \
| sort - tmp > gbToNewName
wc -l gbToNewName
#524911 gbToNewName
# Rename first because we'll need to re-condense nodes with matUtils extract after renaming anyway.
#-- most of those are because of dumb shortened dates, oh well.
# Watch out for creating dups (i.e. new names that are already in tree).
grep -Fwf <(cut -f 2 gb.renaming) samples.$buildDate \
| awk -F\| '{ if ($3 == "") { print $1 "\t" $0; } else { print $2 "\t" $0; } }' \
| sort > gbToNewNameAlreadyInTree
wc -l gbToNewNameAlreadyInTree
#212 gbToNewNameAlreadyInTree
grep -Fwf <(cut -f 1 gbToNewNameAlreadyInTree) gbToOldName \
| cut -f 2 \
| grep -vFwf <(cut -f 2 gbToNewNameAlreadyInTree) \
    > gbOldNameToRemove
wc -l gbOldNameToRemove
#212 gbOldNameToRemove
grep -vFwf <(cut -f 2 gbToNewNameAlreadyInTree) gbToNewName > gbToNewNameNoDup
wc -l gbToNewNameNoDup
#524699 gbToNewNameNoDup
join -t$'\t' gbToOldName gbToNewNameNoDup \
| tawk '$2 != $3 {print $2, $3;}' \
    > gb.renaming
wc -l gb.renaming
#231318 gb.renaming
$matUtils mask -i gisaidAndPublic.$buildDate.masked.noDateDups.pb \
    -r gb.renaming \
    -o gisaidAndPublic.$buildDate.masked.noDateDups.gbRenamed.pb >& renaming.out
# Now remove withdrawn sequences (and re-condense nodes)
comm -23 <(cut -f 1 gbToOldName) <(cut -f 1 gbToNewName) \
| grep -Fwf - gbToOldName \
| cut -f 2 \
    > gbWithdrawnToRemove
wc -l gbWithdrawnToRemove
#5752 gbWithdrawnToRemove
$matUtils extract -i gisaidAndPublic.$buildDate.masked.noDateDups.gbRenamed.pb \
    -p -s <(cat gbOldNameToRemove gbWithdrawnToRemove) \
    -O -o gisaidAndPublic.$buildDate.masked.noDateDups.gbRenamed.noGbWithdrawn.pb
$matUtils summary -i gisaidAndPublic.$buildDate.masked.noDateDups.gbRenamed.noGbWithdrawn.pb
#Total Nodes in Tree: 2328872
#Total Samples in Tree: 1905091
#Total Condensed Nodes in Tree: 229504
#Total Samples in Condensed Nodes: 1209926
#Total Tree Parsimony: 1856952

# OK, one more big rename to replace stupid shortened dates with standardized dates.
$matUtils extract -i gisaidAndPublic.$buildDate.masked.noDateDups.gbRenamed.noGbWithdrawn.pb \
    -u samples.$buildDate
grep -E '\|[0-9]{2}(-[0-9]+)*$' samples.$buildDate \
| sed -re 's/^(.*)\|([0-9]{2}(-[0-9]+)*)$/\1|\2\t\1|20\2/;' \
    > date.renaming
$matUtils mask \
    -i gisaidAndPublic.$buildDate.masked.noDateDups.gbRenamed.noGbWithdrawn.pb \
    -r date.renaming \
    -o gisaidAndPublic.$buildDate.masked.noDateDups.gbRenamed.noGbWithdrawn.stdDates.pb
$matUtils summary -i gisaidAndPublic.$buildDate.masked.noDateDups.gbRenamed.noGbWithdrawn.stdDates.pb
#Total Nodes in Tree: 2328872
#Total Samples in Tree: 1905091
#Total Condensed Nodes in Tree: 0
#Total Samples in Condensed Nodes: 0
#Total Tree Parsimony: 1856952
$matUtils extract \
    -i gisaidAndPublic.$buildDate.masked.noDateDups.gbRenamed.noGbWithdrawn.stdDates.pb \
    -O -o gisaidAndPublic.$buildDate.masked.noDateDups.gbRenamed.noGbWithdrawn.stdDates.recondensed.pb

# Replace gisaidAndPublic.$buildDate.masked.pb with updated protobuf
rm gisaidAndPublic.$buildDate.masked.pb
ln gisaidAndPublic.$buildDate.masked.noDateDups.gbRenamed.noGbWithdrawn.stdDates.recondensed.pb \
    gisaidAndPublic.$buildDate.masked.pb
$matUtils summary -i gisaidAndPublic.$buildDate.masked.pb
#Total Nodes in Tree: 2328872
#Total Samples in Tree: 1905091
#Total Condensed Nodes in Tree: 229504
#Total Samples in Condensed Nodes: 1209926
#Total Tree Parsimony: 1856952
$matUtils extract -i gisaidAndPublic.$buildDate.masked.pb -u samples.$buildDate
sampleCountComma=$(echo $(wc -l < samples.$buildDate) \
                   | sed -re 's/([0-9]+)([0-9]{3})$/\1,\2/; s/([0-9]+)([0-9]{3},[0-9]{3})$/\1,\2/;')
echo "$sampleCountComma genomes from GISAID, GenBank, COG-UK and CNCB ($today); sarscov2phylo 13-11-20 tree with newer sequences added by UShER" \
    > hgPhyloPlace.plusGisaid.description.txt

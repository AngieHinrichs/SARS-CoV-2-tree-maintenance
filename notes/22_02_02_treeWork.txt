# Here's what I did:
cd $ottoDir/2022-02-01
zcat $gisaidDir/chunks/nextclade.full.tsv.gz | $scriptDir/findDropoutContam.pl > gisaid.dropoutContam
zcat $ncbiDir/nextclade.full.tsv.gz | $scriptDir/findDropoutContam.pl > gb.dropoutContam
zcat $cogUkDir/nextclade.full.tsv.gz | $scriptDir/findDropoutContam.pl > cog.dropoutContam
cut -f 1 *.dropoutContam \
| awk -F\| '{ if ($3 == "") { print $1; } else { print $2; } }' \
    > dropoutContam.ids
# Oops, the cluster run was still going.  Work with the latest completed build:

today=2022-01-31
cd $ottoDir/$today
g -Fwf <(cat ../badBranchSeed.ids ../2022-02-01/dropoutContam.ids) samples.$today \
    > samples.dropoutContam
wc -l samples.dropoutContam
#16222 samples.dropoutContam
$matUtils extract -i gisaidAndPublic.$today.masked.pb \
    -p -s samples.dropoutContam \
    -o gisaidAndPublic.$today.masked.dropoutPruned.pb
$matOptimize -T 50 -r 20 -M 1 -S move_log.dropoutPruned \
    -i gisaidAndPublic.$today.masked.dropoutPruned.pb \
    -o gisaidAndPublic.$today.masked.dropoutPruned.opt.pb \
    >& matOptimize.dropoutPruned.log

# Make taxodium
zcat /hive/data/genomes/wuhCor1/goldenPath/bigZips/genes/ncbiGenes.gtf.gz \
| grep -v '"ORF1a"' > ncbiGenes.gtf
zcat /hive/data/genomes/wuhCor1/wuhCor1.fa.gz > wuhCor1.fa
zcat gisaidAndPublic.$today.metadata.tsv.gz > metadata.tmp.tsv
time $matUtils extract -i gisaidAndPublic.$today.masked.dropoutPruned.opt.pb \
    -f wuhCor1.fa \
    -g ncbiGenes.gtf \
    -M metadata.tmp.tsv \
    --extra-fields pango_lineage_usher \
    --write-taxodium gisaidAndPublic.$today.masked.dropoutPruned.opt.taxodium.pb
rm metadata.tmp.tsv wuhCor1.fa
pigz -f -p 8 gisaidAndPublic.$today.masked.dropoutPruned.opt.taxodium.pb

# Mask it and double-optimize.

today=2022-08-23
cd $ottoDir/$today

time $scriptDir/maskDelta.sh merged.pb merged.deltaMasked.pb
#real    15m12.213s

# Prune samples with too many private mutations and internal branches that are too long.
$matUtils extract -i merged.deltaMasked.pb \
    --max-parsimony 20 \
    --max-branch-length 50 \
    --max-path-length 150 \
    -O -o merged.deltaMasked.filtered.pb

# Two rounds of matOptimize since that's necessary after introducing new masking.
time $matOptimize \
    -T 60 -r 8 -M 2 -S move_log.filtered \
    -i merged.deltaMasked.filtered.pb \
    -o merged.deltaMasked.filtered.opt1.pb \
    >& matOptimize.filtered1.log

time $matOptimize \
    -T 60 -r 8 -M 2 -S move_log.filtered \
    -i merged.deltaMasked.filtered.opt1.pb \
    -o gisaidAndPublic.$today.masked.preTrim.pb \
    >& matOptimize.filtered2.log
